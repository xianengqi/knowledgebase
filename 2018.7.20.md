### 我想让如下代码输出“hey amy” , 但却输出了“hey arnode ,这是为什么？

```js
  function greetItem(person) {
    if (person == { name: 'amy'}) {
      return 'hey amy'
    } else {
      return 'hey arnode'
    }
  }

  greetItem({ name: 'amy' })  // hey arnode
```

// 更改后的代码
```js
  function greetItem(person) {
    if (person !== { name: 'amy' }) {    // "!=="用于将表达式的值转换为布尔类型的值，只要类型和值有一样不相等就为true
      return 'hey amy'
    } else {
      return 'hey arnode'
    }

  }
  greetItem({ name: 'amy' }) // hey amy
```
